-- Adicionar table config_ui_usuarios (se não existir)
CREATE TABLE IF NOT EXISTS public.config_ui_usuarios (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
  organizacao_id UUID NOT NULL REFERENCES public.organizacoes(id) ON DELETE CASCADE,
  config_ui JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT now() NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT now() NOT NULL,
  deleted_at TIMESTAMPTZ,
  UNIQUE (user_id, organizacao_id)
);

ALTER TABLE public.config_ui_usuarios ENABLE ROW LEVEL SECURITY;

CREATE POLICY "org_isolation" ON public.config_ui_usuarios
  USING (organizacao_id = public.get_org_id())
  WITH CHECK (organizacao_id = public.get_org_id());

CREATE INDEX idx_config_ui_usuarios_user ON public.config_ui_usuarios(user_id);
CREATE INDEX idx_config_ui_usuarios_org ON public.config_ui_usuarios(organizacao_id);
CREATE INDEX idx_config_ui_usuarios_org_user ON public.config_ui_usuarios(organizacao_id, user_id);

CREATE TRIGGER set_updated_at BEFORE UPDATE ON public.config_ui_usuarios
