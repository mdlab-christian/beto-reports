inferMimetypeFromExtension()

export function inferMimetypeFromExtension(filename: string): string | null {
  const ext = filename.split('.').pop()?.toLowerCase();
  const map: Record<string, string> = {
    'pdf': 'application/pdf', 'doc': 'application/msword',
    'docx': 'application/vnd...wordprocessingml.document',
    'xls': 'application/vnd.ms-excel',
    'xlsx': 'application/vnd...spreadsheetml.sheet',
    'jpg': 'image/jpeg', 'jpeg': 'image/jpeg', 'png': 'image/png',
    'mp4': 'video/mp4', '3gp': 'video/3gpp',
    'aac': 'audio/aac', 'm4a': 'audio/mp4', 'mp3': 'audio/mpeg',
    'ogg': 'audio/ogg', 'opus': 'audio/opus'
  };
  return map[ext || ''] || null;
}

4. Envio de Texto

const handleSend = async (e?: React.FormEvent) => {
  e?.preventDefault();
  if (isComposing || isSendingAttachment) return;
  
  // Se tem anexos pendentes → envia anexos + texto como legenda
  if (pendingAttachments.length > 0) {
    await handleSendMultipleAttachments(inputText.trim());
    setInputText('');
    clearDraft(activeChat?.id);
    return;
  }
  
  // Senão → envia texto puro
  if (!inputText.trim() || !activeChat) return;
  await sendMessage(activeChat.id, inputText.trim());
  setInputText('');
  clearDraft(activeChat.id);
};

Fluxo: handleSend() → sendMessage() (do hook useConversations) → api.sendWhatsappMessage() → Edge Function send-whatsapp-message.

  onDragEnter={(e) => {
    if (!hasDraggedFiles(e.dataTransfer)) return; // ignora drag de texto
    dragCounterRef.current++;     // contador resolve flicker de filhos
    setIsDragOver(true);
  }}
  onDragLeave={(e) => {
    dragCounterRef.current--;
    if (dragCounterRef.current <= 0) setIsDragOver(false);
  }}
  onDrop={(e) => {
    dragCounterRef.current = 0;
    setIsDragOver(false);
    const files = Array.from(e.dataTransfer.files);
    if (files.length > 0) ingestDroppedFiles(files);
  }}
>
